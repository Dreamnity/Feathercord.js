## Creating a simple bot

1. Require the library
```js
// index.js
const Styro = require('styrocord.js');
```
2. Initiating the bot
```diff lang="js"
const Styro = require('styrocord.js');

+ const bot = new Styro({
+     login: {
+         token: 'BOT_TOKEN'
+     }
+ });
```
3. Create `ready` events
```diff lang="js"
const Styro = require('styrocord.js');

const bot = new Styro({
    login: {
        token: 'BOT_TOKEN'
    }
});
+ bot.on("ready",async () => {
+     bot.users["@me"]().then(me=>
+ 	    console.log("Logged in as "+me.username+"!")
+ 	)
+ });
```
4. Create a basic `!ping` command
```diff lang="js"
const Styro = require('styrocord.js');

const bot = new Styro({
    login: {
        token: 'BOT_TOKEN'
    }
});
bot.on("ready",async () => {
    bot.users["@me"]().then(me=>
	    console.log("Logged in as "+me.username+"!")
	)
});
+ bot.on("messageCreate", async msg =>
+     if(msg.content.startsWith('!ping'))
+ 	    return bot.channels[msg.channel.id].messages.create('Pong!')
+ );
```
5. Voila! now you have a working discord bot made with styrocord.js!
